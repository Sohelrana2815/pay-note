import { z } from "zod";

export const invoiceTypes = ["Invoice", "Receipt"] as const;

export const itemSchema = z.object({
  itemName: z.string().nonempty("Item Name is required"),
  itemQuantity: z.number().nonnegative("Item Quantity is required"),
  itemPrice: z.number().nonnegative("Item Price is required"),
});

export const invoiceSchema = z.object({
  type: z.enum(invoiceTypes),

  // Business Detail
  businessName: z
    .string()
    .nonempty("Business Name is required")
    .min(3, "Business Name must be at least 3 characters")
    .max(100, "Business Name cannot exceed 100 characters"),
  logoUrl: z.url("Invalid URL").optional().or(z.literal("")),
  invoiceNumber: z
    .string()
    .nonempty("Invoice Number is required")
    .min(3, "Invoice Number must be at least 3 characters"),
  date: z.date(),
  // Client Detail
  clientName: z
    .string()
    .nonempty("Client Name is required")
    .min(3, "Client Name must be at least 3 characters long")
    .max(100, "Client Name must be at most 100 characters long"),

  clientEmail: z.email("Invalid email").optional().or(z.literal("")),
  clientPhone: z
    .string()
    .nonempty("Client Phone number is required")
    .regex(/^\+8801\d{9}$/, "Invalid Bangladeshi phone number")
    .optional()
    .or(z.literal("")),
  // Items
  items: z
    .array(itemSchema)
    .min(1, "At least one item is required")
    .max(10, "You can add up to 10 items at a time"),
  paymentMethod: z
    .string()
    .nonempty("Payment Method is required")
    .min(3, "Payment Method must be at least 3 characters long")
    .max(100, "Payment Method must be at most 100 characters long"),
  notes: z.string().optional().or(z.literal("")),
});
